1 Make基础

make的标签：第一行为所需文件/标签；第二行为命令 

main: c_math.o test_main.o  # 指定文件的来源，可以是文件或者标签
	clang -I src -o main.exe main.o c_math.o  # 编译命令，必须是已存在的文件

c_math.o : src/c_math.c   # 指定文件的来源
	clang -c src/c_math.c

test_main.o : test/main.c
	clang -I src -c test/main.c
